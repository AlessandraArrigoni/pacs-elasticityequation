91 void parseList( const std::string& slist, std::list<EntryType>& list )
92 {
93  std::string stringList = slist;
94  if ( slist == "" )
95  {
96  return;
97  }
98 
99  int commaPos = 0;
100

102  {
103  commaPos = stringList.find( "," );
104 
105  std::stringstream stream;

107 
108  EntryType var;

110  list.push_back( var );

112  stringList = stringList.substr( commaPos + 1 );
